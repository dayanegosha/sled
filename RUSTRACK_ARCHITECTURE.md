### Key Architectural Decisions

| Concern | Decision | Rationale |
|---|---|---|
| SSR vs SPA | Next.js App Router (hybrid) | SEO for public pages, client components for map |
| State management | Zustand + React Query | Lightweight, server-state separation |
| Real-time | Socket.IO over WebSocket | Reconnection logic, rooms, namespaces |
| Geo storage | PostGIS Geography type | Accurate spherical distance, ST_Union for merge |
| Reveal overlay | Mapbox custom layer (WebGL) | Hardware-accelerated, smooth animation |
| Heatmap | PostGIS grid aggregation → Mapbox | Server-computed tiles, cached in Redis |
| Auth | VK OAuth → httpOnly JWT cookie | No localStorage tokens, XSS-safe |

## 3. Database Schema

```sql
-- ============================================================
-- MIGRATION: 001_init.sql
-- ============================================================

-- Enable PostGIS
CREATE EXTENSION IF NOT EXISTS postgis;
CREATE EXTENSION IF NOT EXISTS postgis_topology;

-- ─────────────────────────────────────────────────────────────
-- USERS
-- ─────────────────────────────────────────────────────────────
CREATE TABLE users (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    vk_id           BIGINT UNIQUE NOT NULL,
    username        VARCHAR(64) NOT NULL,
    display_name    VARCHAR(128) NOT NULL,
    avatar_url      TEXT,
    bio             TEXT,
    is_admin        BOOLEAN NOT NULL DEFAULT FALSE,
    is_banned       BOOLEAN NOT NULL DEFAULT FALSE,
    ban_reason      TEXT,
    total_distance  DOUBLE PRECISION NOT NULL DEFAULT 0,   -- meters
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    last_seen_at    TIMESTAMPTZ
);

CREATE INDEX idx_users_vk_id ON users(vk_id);
CREATE INDEX idx_users_is_banned ON users(is_banned) WHERE is_banned = FALSE;

-- ─────────────────────────────────────────────────────────────
-- TRACKS (raw GPS points, append-only time-series)
-- ─────────────────────────────────────────────────────────────
CREATE TABLE tracks (
    id          BIGSERIAL PRIMARY KEY,
    user_id     UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    -- GEOGRAPHY type uses WGS84, distance in meters
    location    GEOGRAPHY(POINT, 4326) NOT NULL,
    altitude    DOUBLE PRECISION,           -- meters above sea level
    accuracy    DOUBLE PRECISION,           -- GPS accuracy in meters
    speed       DOUBLE PRECISION,           -- m/s
    heading     DOUBLE PRECISION,           -- degrees
    recorded_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- BRIN index for time-series scans; GiST for spatial queries
CREATE INDEX idx_tracks_user_id ON tracks(user_id);
CREATE INDEX idx_tracks_recorded_at ON tracks USING BRIN(recorded_at);
CREATE INDEX idx_tracks_location ON tracks USING GIST(location);
CREATE INDEX idx_tracks_user_location ON tracks USING GIST(location)
    WHERE user_id IS NOT NULL;

-- ─────────────────────────────────────────────────────────────
-- REVEALED AREAS (merged polygon per user, the "fog of war" mask)
-- ─────────────────────────────────────────────────────────────
CREATE TABLE revealed_areas (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id     UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    -- Merged MultiPolygon of all revealed circles
    geom        GEOGRAPHY(MULTIPOLYGON, 4326) NOT NULL,
    area_m2     DOUBLE PRECISION NOT NULL DEFAULT 0,
    updated_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE UNIQUE INDEX idx_revealed_areas_user ON revealed_areas(user_id);
CREATE INDEX idx_revealed_areas_geom ON revealed_areas USING GIST(geom);

-- ─────────────────────────────────────────────────────────────
-- RUSSIAN REGIONS (reference data — seeded separately)
-- ─────────────────────────────────────────────────────────────
CREATE TABLE regions (
    id          SERIAL PRIMARY KEY,
    name_ru     VARCHAR(128) NOT NULL,
    name_en     VARCHAR(128) NOT NULL,
    name_zh     VARCHAR(128),
    code        VARCHAR(16) UNIQUE NOT NULL,  -- ISO 3166-2:RU
    geom        GEOGRAPHY(MULTIPOLYGON, 4326) NOT NULL,
    area_m2     DOUBLE PRECISION NOT NULL
);

CREATE INDEX idx_regions_geom ON regions USING GIST(geom);

-- ─────────────────────────────────────────────────────────────
-- USER REGION EXPLORATION STATS (materialized per region)
-- ─────────────────────────────────────────────────────────────
CREATE TABLE user_region_stats (
    user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    region_id       INTEGER NOT NULL REFERENCES regions(id),
    explored_m2     DOUBLE PRECISION NOT NULL DEFAULT 0,
    explored_pct    DOUBLE PRECISION NOT NULL DEFAULT 0,  -- 0.0 - 100.0
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    PRIMARY KEY (user_id, region_id)
);

CREATE INDEX idx_urs_user ON user_region_stats(user_id);
CREATE INDEX idx_urs_explored ON user_region_stats(user_id, explored_pct DESC);

-- ─────────────────────────────────────────────────────────────
-- POSTS (community feed)
-- ─────────────────────────────────────────────────────────────
CREATE TABLE posts (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    content         TEXT NOT NULL CHECK (char_length(content) <= 2000),
    image_urls      TEXT[] DEFAULT '{}',
    location        GEOGRAPHY(POINT, 4326),
    location_name   VARCHAR(256),
    likes_count     INTEGER NOT NULL DEFAULT 0,
    comments_count  INTEGER NOT NULL DEFAULT 0,
    is_hidden       BOOLEAN NOT NULL DEFAULT FALSE,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_posts_user_id ON posts(user_id);
CREATE INDEX idx_posts_created_at ON posts(created_at DESC);
CREATE INDEX idx_posts_location ON posts USING GIST(location)
    WHERE location IS NOT NULL;

-- ─────────────────────────────────────────────────────────────
-- COMMENTS
-- ─────────────────────────────────────────────────────────────
CREATE TABLE comments (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    post_id     UUID NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
    user_id     UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    content     TEXT NOT NULL CHECK (char_length(content) <= 500),
    is_hidden   BOOLEAN NOT NULL DEFAULT FALSE,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_comments_post_id ON comments(post_id, created_at);
CREATE INDEX idx_comments_user_id ON comments(user_id);

-- ─────────────────────────────────────────────────────────────
-- LIKES (posts only; extend for comments if needed)
-- ─────────────────────────────────────────────────────────────
CREATE TABLE likes (
    user_id     UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    post_id     UUID NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    PRIMARY KEY (user_id, post_id)
);

CREATE INDEX idx_likes_post_id ON likes(post_id);

-- Trigger: keep posts.likes_count in sync
CREATE OR REPLACE FUNCTION sync_likes_count()
RETURNS TRIGGER AS $$
BEGIN
  IF TG_OP = 'INSERT' THEN
    UPDATE posts SET likes_count = likes_count + 1
    WHERE id = NEW.post_id;
  ELSIF TG_OP = 'DELETE' THEN
    UPDATE posts SET likes_count = likes_count - 1
    WHERE id = OLD.post_id;
  END IF;
  RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_sync_likes_count
AFTER INSERT OR DELETE ON likes
FOR EACH ROW EXECUTE FUNCTION sync_likes_count();

-- ─────────────────────────────────────────────────────────────
-- FRIENDSHIPS / FOLLOWS
-- ─────────────────────────────────────────────────────────────
CREATE TABLE friendships (
    requester_id    UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    addressee_id    UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    status          VARCHAR(16) NOT NULL DEFAULT 'pending'
                    CHECK (status IN ('pending', 'accepted', 'blocked')),
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    PRIMARY KEY (requester_id, addressee_id),
    CHECK (requester_id <> addressee_id)
);

CREATE INDEX idx_friendships_addressee ON friendships(addressee_id, status);
CREATE INDEX idx_friendships_requester ON friendships(requester_id, status);

-- ─────────────────────────────────────────────────────────────
-- HEATMAP GRID CACHE (pre-aggregated, refreshed by cron)
-- ─────────────────────────────────────────────────────────────
CREATE TABLE heatmap_grid (
    cell_id     BIGSERIAL PRIMARY KEY,
    -- Grid cell centroid
    location    GEOGRAPHY(POINT, 4326) NOT NULL,
    -- Number of unique users who visited this cell
    user_count  INTEGER NOT NULL DEFAULT 0,
    -- Total visits (non-unique)
    visit_count INTEGER NOT NULL DEFAULT 0,
    zoom_level  SMALLINT NOT NULL DEFAULT 10,
    updated_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_heatmap_location ON heatmap_grid USING GIST(location);
CREATE INDEX idx_heatmap_zoom ON heatmap_grid(zoom_level);

-- ─────────────────────────────────────────────────────────────
-- REFRESH MATERIALIZED VIEW FOR HEATMAP
-- ─────────────────────────────────────────────────────────────
CREATE MATERIALIZED VIEW heatmap_zoom10 AS
SELECT
    ST_SnapToGrid(ST_GeomFromEWKB(location::bytea), 0.01) AS cell_center,
    COUNT(DISTINCT user_id) AS user_count,
    COUNT(*) AS visit_count
FROM tracks
GROUP BY cell_center;

CREATE UNIQUE INDEX ON heatmap_zoom10(cell_center);

-- ─────────────────────────────────────────────────────────────
-- ADMIN AUDIT LOG
-- ─────────────────────────────────────────────────────────────
CREATE TABLE admin_audit_log (
    id          BIGSERIAL PRIMARY KEY,
    admin_id    UUID NOT NULL REFERENCES users(id),
    action      VARCHAR(64) NOT NULL,
    target_type VARCHAR(32),  -- 'user', 'post', 'comment'
    target_id   TEXT,
    metadata    JSONB,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_audit_admin ON admin_audit_log(admin_id, created_at DESC);
```

## 4. API Design

### Base URL: `/api/v1`

All protected routes require `Authorization: Bearer <token>` header (set automatically from httpOnly cookie via middleware).

#### Auth Module

```
POST   /auth/vk                → Redirect to VK OAuth page
GET    /auth/vk/callback        → Handle VK code, set httpOnly cookie, redirect to /map
POST   /auth/logout             → Clear cookie
GET    /auth/me                 → Return current user from token
POST   /auth/refresh            → Refresh JWT (if using refresh token pattern)
```

#### Users Module

```
GET    /users/:id               → Public profile
GET    /users/:id/stats         → Exploration stats
GET    /users/:id/posts         → User's posts (paginated)
PATCH  /users/me                → Update profile (bio, avatar)
```

#### Tracks Module

```
POST   /tracks/batch            → Upload batch of GPS points (max 100 per request)
GET    /tracks/revealed         → Get user's revealed area as GeoJSON
GET    /tracks/stats            → Distance, % explored Russia, top regions
GET    /tracks/regions          → Per-region exploration stats

WebSocket: ws://api/tracks
  emit: 'location'  { lat, lng, accuracy, timestamp }
  on:   'area_updated' { newGeojson, distanceDelta }
```

#### Community Module

```
GET    /posts                   → Feed (following + popular), cursor pagination
POST   /posts                   → Create post (multipart/form-data)
GET    /posts/:id               → Single post with comments
DELETE /posts/:id               → Delete own post
POST   /posts/:id/like          → Toggle like
DELETE /posts/:id/like
GET    /posts/:id/comments      → Comments (paginated)
POST   /posts/:id/comments      → Add comment
DELETE /comments/:id            → Delete own comment
```

#### Friendships Module

```
GET    /friendships             → My accepted friends list
GET    /friendships/requests    → Incoming pending requests
POST   /friendships/:userId     → Send friend request
PATCH  /friendships/:userId     → Accept / reject (body: { action: 'accept'|'reject' })
DELETE /friendships/:userId     → Unfriend / cancel
GET    /friendships/compare/:userId → Compare stats with friend
```

#### Heatmap Module

```
GET    /heatmap                 → GeoJSON FeatureCollection for heatmap
       ?bbox=minLng,minLat,maxLng,maxLat
       &zoom=10
       (Redis cached per bbox+zoom, TTL 5min)
```

#### Admin Module (requires `is_admin = true`)

```
GET    /admin/stats             → Platform-wide analytics
GET    /admin/users             → Paginated user list (search, filter)
PATCH  /admin/users/:id/ban     → Ban user { reason: string }
PATCH  /admin/users/:id/unban
GET    /admin/posts             → All posts (with hidden ones)
```

```typescript
import { Injectable, UnauthorizedException } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import axios from 'axios';
import { User } from '../users/entities/user.entity';

@Injectable()
export class AuthService {
  constructor(
    @InjectRepository(User)
    private readonly usersRepo: Repository<User>,
    private readonly jwtService: JwtService,
  ) {}

  async handleVkCallback(code: string) {
    // 1. Exchange code for VK access token
    const tokenRes = await axios.get('https://oauth.vk.com/access_token', {
      params: {
        client_id:     process.env.VK_APP_ID,
        client_secret: process.env.VK_APP_SECRET,
        redirect_uri:  process.env.VK_CALLBACK_URL,
        code,
      },
    });

    const { access_token, user_id, email } = tokenRes.data;
    if (!access_token) throw new UnauthorizedException('VK token exchange failed');

    // 2. Fetch VK user profile
    const profileRes = await axios.get('https://api.vk.com/method/users.get', {
      params: {
        user_ids: user_id,
        fields:   'photo_200,screen_name',
        v:        '5.199',
        access_token,
      },
    });

    const vkUser = profileRes.data.response[0];

    // 3. Upsert user in our DB
    let user = await this.usersRepo.findOne({ where: { vkId: user_id } });
    if (!user) {
      user = this.usersRepo.create({
        vkId:        user_id,
        username:    vkUser.screen_name || `user_${user_id}`,
        displayName: `${vkUser.first_name} ${vkUser.last_name}`,
        avatarUrl:   vkUser.photo_200,
      });
    } else {
      user.avatarUrl   = vkUser.photo_200;
      user.displayName = `${vkUser.first_name} ${vkUser.last_name}`;
    }
    await this.usersRepo.save(user);

    // 4. Issue our own JWT pair
    return this.generateTokenPair(user);
  }

  generateTokenPair(user: User) {
    const payload = { sub: user.id, isAdmin: user.isAdmin };
    return {
      accessToken: this.jwtService.sign(payload, {
        secret:    process.env.JWT_SECRET,
        expiresIn: '15m',
      }),
      refreshToken: this.jwtService.sign(payload, {
        secret:    process.env.JWT_REFRESH_SECRET,
        expiresIn: '30d',
      }),
    };
  }

  async refreshTokens(refreshToken: string) {
    try {
      const payload = this.jwtService.verify(refreshToken, {
        secret: process.env.JWT_REFRESH_SECRET,
      });
      const user = await this.usersRepo.findOneOrFail({ where: { id: payload.sub } });
      if (user.isBanned) throw new UnauthorizedException('User is banned');
      return this.generateTokenPair(user);
    } catch {
      throw new UnauthorizedException('Invalid refresh token');
    }
  }
```

## 7. Reveal Logic

### Core Concept

Each GPS point → circle polygon (20m radius) → union with user's existing revealed area → store merged MultiPolygon in PostGIS → send updated GeoJSON to client → update Mapbox `reveal-mask` layer to cut hole in fog overlay.

### Backend: `tracks.service.ts`

```typescript
// src/modules/tracks/tracks.service.ts
import { Injectable } from '@nestjs/common';
import { InjectDataSource } from '@nestjs/typeorm';
import { DataSource } from 'typeorm';
import { Redis } from 'ioredis';
import { InjectRedis } from '@nestjs-modules/ioredis';
import { BatchTrackDto } from './dto/batch-track.dto';

const REVEAL_RADIUS_METERS = 20;

@Injectable()
export class TracksService {
  constructor(
    @InjectDataSource() private readonly db: DataSource,
    @InjectRedis() private readonly redis: Redis,
  ) {}

  /**
   * Process a batch of GPS points from a user.
   * Returns the delta (new area added) as GeoJSON.
   */
  async processBatch(userId: string, points: BatchTrackDto[]) {
    if (!points.length) return null;

    const result = await this.db.transaction(async (em) => {
      // 1. Insert raw track points
      const values = points
        .map((_, i) => `($${i * 4 + 1}, $${i * 4 + 2}, $${i * 4 + 3}::geography, $${i * 4 + 4})`)
        .join(', ');

      const flatParams = points.flatMap((p) => [
        userId,
        p.timestamp,
        `SRID=4326;POINT(${p.lng} ${p.lat})`,
        p.accuracy ?? null,
      ]);

      await em.query(
        `INSERT INTO tracks (user_id, recorded_at, location, accuracy) VALUES ${values}`,
        flatParams,
      );

      // 2. Build union of new circles using PostGIS
      const circleUnion = points
        .map((_, i) => `ST_Buffer(ST_GeogFromText($${i * 2 + 1})::geometry, ${REVEAL_RADIUS_METERS})`)
        .join(', ');

      const newCirclesParams = points.flatMap((p) => [
        `SRID=4326;POINT(${p.lng} ${p.lat})`,
        REVEAL_RADIUS_METERS,
      ]);

      // 3. Merge with existing revealed area and compute delta
      const mergeResult = await em.query(
        `
        WITH new_circles AS (
          SELECT ST_Union(ARRAY[${points
            .map((p) => `ST_Buffer(ST_GeogFromText('SRID=4326;POINT(${p.lng} ${p.lat})')::geometry, ${REVEAL_RADIUS_METERS})`)
            .join(', ')}]) AS geom
        ),
        existing AS (
          SELECT COALESCE(geom::geometry, ST_GeomFromText('GEOMETRYCOLLECTION EMPTY', 4326)) AS geom
          FROM revealed_areas
          WHERE user_id = $1
          FOR UPDATE
        ),
        merged AS (
          SELECT
            ST_Union(e.geom, n.geom) AS merged_geom,
            ST_Difference(n.geom, e.geom) AS delta_geom,
            ST_Area(ST_Union(e.geom, n.geom)::geography) AS new_area
          FROM existing e, new_circles n
        )
        INSERT INTO revealed_areas (user_id, geom, area_m2)
        SELECT $1, merged_geom::geography, new_area FROM merged
        ON CONFLICT (user_id) DO UPDATE
          SET geom     = EXCLUDED.geom,
              area_m2  = EXCLUDED.area_m2,
              updated_at = NOW()
        RETURNING
          area_m2,
          ST_AsGeoJSON((SELECT delta_geom FROM merged))::json AS delta_geojson
        `,
        [userId],
      );

      return mergeResult[0];
    });

    if (!result) return null;

    // 4. Update user total distance
    await this.updateDistance(userId, points);

    // 5. Invalidate Redis cache for this user's GeoJSON
    await this.redis.del(`revealed:${userId}`);

    // 6. Update region stats asynchronously (fire-and-forget)
    this.updateRegionStats(userId).catch(console.error);

    return {
      deltaGeoJSON: result.delta_geojson,
      totalAreaM2:  result.area_m2,
    };
  }

  /**
   * Get full revealed area as GeoJSON FeatureCollection (with Redis caching)
   */
  async getRevealedArea(userId: string) {
    const cacheKey = `revealed:${userId}`;
    const cached = await this.redis.get(cacheKey);
    if (cached) return JSON.parse(cached);

    const rows = await this.db.query(
      `SELECT ST_AsGeoJSON(geom)::json AS geojson, area_m2
       FROM revealed_areas WHERE user_id = $1`,
      [userId],
    );

    if (!rows.length) {
      return { type: 'FeatureCollection', features: [] };
    }

    const result = {
      type: 'FeatureCollection',
      features: [{
        type:       'Feature',
        geometry:   rows[0].geojson,
        properties: { area_m2: rows[0].area_m2 },
      }],
    };

    // Cache for 30 seconds
    await this.redis.setex(cacheKey, 30, JSON.stringify(result));
    return result;
  }

  /**
   * Calculate distance from batch of ordered GPS points
   */
  private async updateDistance(userId: string, points: BatchTrackDto[]) {
    if (points.length < 2) return;

    const distQuery = await this.db.query(
      `
      WITH pts AS (
        SELECT unnest($1::geometry[]) AS geom
      ),
      numbered AS (
        SELECT geom, ROW_NUMBER() OVER () AS rn FROM pts
      ),
      distances AS (
        SELECT ST_Distance(a.geom::geography, b.geom::geography) AS d
        FROM numbered a
        JOIN numbered b ON b.rn = a.rn + 1
      )
      SELECT COALESCE(SUM(d), 0) AS total_dist FROM distances
      `,
      [points.map((p) => `SRID=4326;POINT(${p.lng} ${p.lat})`).join(',')],
    );

    const delta = distQuery[0]?.total_dist ?? 0;
    if (delta > 0) {
      await this.db.query(
        `UPDATE users SET total_distance = total_distance + $1 WHERE id = $2`,
        [delta, userId],
      );
    }
  }

  /**
   * Per-region exploration stats — runs async after each batch
   */
  private async updateRegionStats(userId: string) {
    await this.db.query(
      `
      INSERT INTO user_region_stats (user_id, region_id, explored_m2, explored_pct)
      SELECT
        $1 AS user_id,
        r.id AS region_id,
        ST_Area(ST_Intersection(ra.geom::geometry, r.geom::geometry)::geography) AS explored_m2,
        ROUND(
          ST_Area(ST_Intersection(ra.geom::geometry, r.geom::geometry)::geography)
          / r.area_m2 * 100, 4
        ) AS explored_pct
      FROM revealed_areas ra
      CROSS JOIN regions r
      WHERE ra.user_id = $1
        AND ST_Intersects(ra.geom::geometry, r.geom::geometry)
      ON CONFLICT (user_id, region_id) DO UPDATE
        SET explored_m2  = EXCLUDED.explored_m2,
            explored_pct = EXCLUDED.explored_pct,
            updated_at   = NOW()
      `,
      [userId],
    );
  }
}
```

### Frontend: `useRevealEngine.ts`

```typescript
// src/hooks/useRevealEngine.ts
import { useCallback, useRef } from 'react';
import { useMapbox } from './useMapbox';
import { useTrackStore } from '@/stores/trackStore';
import * as turf from '@turf/turf';
import { apiClient } from '@/services/api';
import type { Map } from 'mapbox-gl';

const REVEAL_RADIUS_M = 20;
const BATCH_SIZE      = 10;  // flush every N points
const BATCH_INTERVAL  = 5000; // or every 5 seconds

const SOURCE_ID = 'reveal-source';
const LAYER_ID  = 'reveal-fill';

/**
 * Manages the client-side reveal overlay.
 *
 * Strategy:
 * 1. Buffer incoming GPS points into a local batch
 * 2. Every BATCH_INTERVAL ms OR BATCH_SIZE points, flush to server
 * 3. Merge server response (delta GeoJSON) into local GeoJSON state
 * 4. Update Mapbox source → layer re-renders
 *
 * The "fog of war" effect is achieved by:
 * - A full-Russia background layer with opacity 0.85 (fog)
 * - A fill-extrusion or fill layer from the revealed MultiPolygon
 *   using 'destination-out' composite operation in a custom WebGL layer
 */
export function useRevealEngine() {
  const { map } = useMapbox();
  const { addPoint, pendingPoints, clearPending } = useTrackStore();
  const localGeoJSON = useRef<GeoJSON.FeatureCollection>({
    type:     'FeatureCollection',
    features: [],
  });
  const flushTimer = useRef<ReturnType<typeof setTimeout>>();

  /**
   * Initialize Mapbox source + layer on first call
   */
  const initMapLayers = useCallback((mapInstance: Map) => {
    if (mapInstance.getSource(SOURCE_ID)) return;

    mapInstance.addSource(SOURCE_ID, {
      type: 'geojson',
      data: localGeoJSON.current,
    });

    // The revealed area punch-through layer
    // Uses 'screen' blend to brighten the base map beneath it
    mapInstance.addLayer({
      id:     LAYER_ID,
      type:   'fill',
      source: SOURCE_ID,
      paint: {
        'fill-color':   '#ffffff',
        'fill-opacity': 0,
      },
    }, 'fog-overlay'); // Insert BELOW the fog layer

    // Counterpart: remove fog ONLY where revealed
    mapInstance.setPaintProperty('fog-overlay', 'background-opacity', [
      'case',
      ['boolean', true, false],
      0.85,
      0.85,
    ]);
  }, []);

  /**
   * Add a new GPS point to the reveal system
   */
  const addRevealPoint = useCallback(
    async (lat: number, lng: number, accuracy: number, timestamp: number) => {
      if (!map) return;

      initMapLayers(map);

      // Immediate client-side preview: add circle to local GeoJSON
      const circle = turf.circle([lng, lat], REVEAL_RADIUS_M / 1000, {
        steps: 32,
        units: 'kilometers',
      });

      const existing = localGeoJSON.current.features[0]?.geometry;
      if (existing) {
        const merged = turf.union(
          turf.featureCollection([
            turf.feature(existing as GeoJSON.Polygon | GeoJSON.MultiPolygon),
            circle,
          ]),
        );
        if (merged) {
          localGeoJSON.current = turf.featureCollection([merged]);
        }
      } else {
        localGeoJSON.current = turf.featureCollection([circle]);
      }

      // Update Mapbox source instantly (smooth feel)
      (map.getSource(SOURCE_ID) as mapboxgl.GeoJSONSource)?.setData(
        localGeoJSON.current,
      );

      // Buffer point for server flush
      addPoint({ lat, lng, accuracy, timestamp });

      // Schedule flush
      if (!flushTimer.current) {
        flushTimer.current = setTimeout(() => flushToServer(), BATCH_INTERVAL);
      }

      if (pendingPoints.length >= BATCH_SIZE) {
        clearTimeout(flushTimer.current);
        flushTimer.current = undefined;
        await flushToServer();
      }
    },
    [map, pendingPoints],
  );

  /**
   * Send buffered points to backend, reconcile with authoritative GeoJSON
   */
  const flushToServer = useCallback(async () => {
    const points = useTrackStore.getState().pendingPoints;
    if (!points.length) return;
    clearPending();

    try {
      const response = await apiClient.post('/tracks/batch', { points });
      const { deltaGeoJSON, totalAreaM2 } = response.data;

      // Server is source of truth — update local cache
      if (deltaGeoJSON && map) {
        const delta = turf.feature(deltaGeoJSON);
        const existing = localGeoJSON.current.features[0]?.geometry;
        if (existing) {
          const merged = turf.union(
            turf.featureCollection([
              turf.feature(existing as any),
              delta,
            ]),
          );
          if (merged) localGeoJSON.current = turf.featureCollection([merged]);
        }
        (map.getSource(SOURCE_ID) as mapboxgl.GeoJSONSource)?.setData(
          localGeoJSON.current,
        );
      }
    } catch (err) {
      console.error('[RevealEngine] flush failed:', err);
      // Points are lost from buffer but not from server (they were already inserted)
      // Client will re-sync on next load
    }
  }, [map]);

  /**
   * Load existing revealed area from server on app start
   */
  const loadRevealedArea = useCallback(async () => {
    if (!map) return;
    initMapLayers(map);

    try {
      const response = await apiClient.get('/tracks/revealed');
      localGeoJSON.current = response.data;
      (map.getSource(SOURCE_ID) as mapboxgl.GeoJSONSource)?.setData(
        localGeoJSON.current,
      );
    } catch (err) {
      console.error('[RevealEngine] load failed:', err);
    }
  }, [map]);

  return { addRevealPoint, loadRevealedArea, flushToServer };
}
```

```typescript
// src/modules/heatmap/heatmap.service.ts
import { Injectable } from '@nestjs/common';
import { InjectDataSource } from '@nestjs/typeorm';
import { DataSource } from 'typeorm';
import { Redis } from 'ioredis';
import { InjectRedis } from '@nestjs-modules/ioredis';

@Injectable()
export class HeatmapService {
  constructor(
    @InjectDataSource() private readonly db: DataSource,
    @InjectRedis() private readonly redis: Redis,
  ) {}

  /**
   * Returns heatmap as GeoJSON FeatureCollection.
   * Cells aggregated at ~1km grid (0.01° ≈ 1.1km at equator).
   * Results cached in Redis per bbox+zoom.
   */
  async getHeatmap(bbox?: string, zoom: number = 10) {
    const cacheKey = `heatmap:${bbox ?? 'world'}:${zoom}`;
    const cached = await this.redis.get(cacheKey);
    if (cached) return JSON.parse(cached);

    let bboxFilter = '';
    const params: any[] = [];

    if (bbox) {
      const [minLng, minLat, maxLng, maxLat] = bbox.split(',').map(Number);
      bboxFilter = `
        AND ST_Within(
          location::geometry,
          ST_MakeEnvelope($1, $2, $3, $4, 4326)
        )
      `;
      params.push(minLng, minLat, maxLng, maxLat);
    }

    // Grid resolution based on zoom level
    const gridSize = zoom >= 13 ? 0.001 : zoom >= 10 ? 0.01 : 0.1;

    const rows = await this.db.query(
      `
      SELECT
        ST_AsGeoJSON(
          ST_SetSRID(
            ST_Point(
              ROUND(ST_X(location::geometry) / ${gridSize}) * ${gridSize},
              ROUND(ST_Y(location::geometry) / ${gridSize}) * ${gridSize}
            ),
            4326
          )
        )::json AS geometry,
        COUNT(DISTINCT user_id)::int AS user_count,
        COUNT(*)::int AS visit_count
      FROM tracks
      WHERE 1=1 ${bboxFilter}
      GROUP BY
        ROUND(ST_X(location::geometry) / ${gridSize}),
        ROUND(ST_Y(location::geometry) / ${gridSize})
      HAVING COUNT(*) > 2
      ORDER BY visit_count DESC
      LIMIT 50000
      `,
      params,
    );

    const result = {
      type: 'FeatureCollection',
      features: rows.map((row: any) => ({
        type:       'Feature',
        geometry:   row.geometry,
        properties: {
          user_count:  row.user_count,
          visit_count: row.visit_count,
          // Normalize weight 0–1 for Mapbox heatmap intensity
          weight: Math.min(1, row.visit_count / 1000),
        },
      })),
    };

    // Cache 5 minutes for bbox queries, 1 min for world view
    const ttl = bbox ? 300 : 60;
    await this.redis.setex(cacheKey, ttl, JSON.stringify(result));

    return result;
  }

  /**
   * Called by cron job — refresh materialized view and invalidate cache
   */
  async refreshHeatmapCache() {
    await this.db.query(`REFRESH MATERIALIZED VIEW CONCURRENTLY heatmap_zoom10`);
    const keys = await this.redis.keys('heatmap:*');
    if (keys.length) await this.redis.del(...keys);
  }
}
```

```typescript
// src/components/map/HeatmapLayer.tsx
'use client';

import { useEffect, useRef } from 'react';
import type { Map } from 'mapbox-gl';
import { apiClient } from '@/services/api';

const HEATMAP_SOURCE = 'heatmap-source';
const HEATMAP_LAYER  = 'heatmap-layer';

interface Props {
  map: Map;
}

export function HeatmapLayer({ map }: Props) {
  const abortRef = useRef<AbortController>();

  useEffect(() => {
    if (!map) return;

    const initHeatmap = async () => {
      try {
        const data = await apiClient.get('/heatmap');

        if (!map.getSource(HEATMAP_SOURCE)) {
          map.addSource(HEATMAP_SOURCE, {
            type: 'geojson',
            data: data.data,
          });
        }

        if (!map.getLayer(HEATMAP_LAYER)) {
          map.addLayer({
            id:     HEATMAP_LAYER,
            type:   'heatmap',
            source: HEATMAP_SOURCE,
            maxzoom: 15,
            paint: {
              // Increase weight as visit_count increases
              'heatmap-weight': [
                'interpolate', ['linear'],
                ['get', 'visit_count'],
                0, 0,
                1000, 1,
              ],
              // Increase intensity as zoom level increases
              'heatmap-intensity': [
                'interpolate', ['linear'], ['zoom'],
                0, 1,
                15, 3,
              ],
              // Assign color values be applied to each pixel based on density
              // blue → cyan → yellow → red (Strava-style)
              'heatmap-color': [
                'interpolate', ['linear'],
                ['heatmap-density'],
                0,    'rgba(13, 30, 60, 0)',
                0.1,  'rgba(0, 100, 255, 0.4)',
                0.3,  'rgba(0, 200, 255, 0.6)',
                0.5,  'rgba(100, 255, 100, 0.7)',
                0.7,  'rgba(255, 200, 0, 0.8)',
                0.9,  'rgba(255, 100, 0, 0.9)',
                1.0,  'rgba(255, 0, 0, 1)',
              ],
              // Adjust the heatmap radius by zoom level
              'heatmap-radius': [
                'interpolate', ['linear'], ['zoom'],
                0, 2,
                9, 20,
                15, 40,
              ],
              // Transition from heatmap to circle layer by zoom level
              'heatmap-opacity': [
                'interpolate', ['linear'], ['zoom'],
                12, 1,
                15, 0,
              ],
            },
          });
        }
      } catch (err) {
        console.error('[HeatmapLayer] failed to load:', err);
      }
    };

    map.on('load', initHeatmap);

    // Refresh heatmap data when map moves (with debounce)
    let debounce: ReturnType<typeof setTimeout>;
    const onMoveEnd = () => {
      clearTimeout(debounce);
      debounce = setTimeout(async () => {
        const bounds = map.getBounds();
        const bbox = [
          bounds.getWest(), bounds.getSouth(),
          bounds.getEast(), bounds.getNorth(),
        ].join(',');
        const zoom = Math.floor(map.getZoom());

        abortRef.current?.abort();
        abortRef.current = new AbortController();

        try {
          const res = await apiClient.get(
            `/heatmap?bbox=${bbox}&zoom=${zoom}`,
            { signal: abortRef.current.signal },
          );
          (map.getSource(HEATMAP_SOURCE) as mapboxgl.GeoJSONSource)
            ?.setData(res.data);
        } catch {
          // Aborted — ignore
        }
      }, 500);
    };

    map.on('moveend', onMoveEnd);

    return () => {
      clearTimeout(debounce);
      abortRef.current?.abort();
      map.off('moveend', onMoveEnd);
      if (map.getLayer(HEATMAP_LAYER)) map.removeLayer(HEATMAP_LAYER);
      if (map.getSource(HEATMAP_SOURCE)) map.removeSource(HEATMAP_SOURCE);
    };
  }, [map]);

  return null;
}
```

```typescript
@Injectable()
export class RateLimitMiddleware implements NestMiddleware {
  constructor(@InjectRedis() private readonly redis: Redis) {}

  async use(req: Request, res: Response, next: NextFunction) {
    const key = `rl:${req.ip}:${req.path}`;

    // Per-route limits
    const limits: Record<string, { max: number; window: number }> = {
      '/auth/vk':         { max: 10,   window: 60 },
      '/tracks/batch':    { max: 120,  window: 60 },  // 2/sec
      '/posts':           { max: 30,   window: 60 },
      '/posts/:id/like':  { max: 60,   window: 60 },
      default:            { max: 300,  window: 60 },
    };

    const matchedRoute = Object.keys(limits).find((k) =>
      req.path.startsWith(k.replace(':id', ''))
    );
    const limit = limits[matchedRoute ?? 'default'];

    const current = await this.redis.incr(key);
    if (current === 1) {
      await this.redis.expire(key, limit.window);
    }

    if (current > limit.max) {
      res.setHeader('Retry-After', limit.window.toString());
      throw new HttpException('Too Many Requests', HttpStatus.TOO_MANY_REQUESTS);
    }

    res.setHeader('X-RateLimit-Limit',     limit.max.toString());
    res.setHeader('X-RateLimit-Remaining', Math.max(0, limit.max - current).toString());

    next();
  }
}
```

## Summary: Key Design Decisions

| Decision | Approach | Why |
|---|---|---|
| **Fog of war** | PostGIS ST_Union + Mapbox WebGL layer | Server-authoritative, hardware-accelerated |
| **GPS batching** | 10 points or 5s interval | Balance between accuracy and battery/bandwidth |
| **Heatmap** | Grid aggregation in PostGIS, Redis tile cache | No third-party tile server needed |
| **Auth** | VK OAuth → httpOnly JWT cookie | XSS-safe, no localStorage exposure |
| **i18n** | react-i18next + JSON locale files | Industry standard, lazy-load per language |
| **PWA** | next-pwa + manifest + service worker | Mapbox tiles cached offline |
| **Real-time** | Socket.IO WebSocket | Live position sharing (future feature) |
| **Admin** | Separate route group, admin guard | Clear separation, same codebase |
